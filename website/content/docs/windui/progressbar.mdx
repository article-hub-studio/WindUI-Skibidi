## ProgressBar Element

Creates a progress bar for showing completion, loading state, or a numeric value.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="indeterminate">Indeterminate</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Progress = Tab:ProgressBar({
    Title = "Download",
    Desc = "Downloading files...",
    Value = {
        Min = 0,
        Max = 100,
        Default = 25,
    },
})
```
    </TabsContent>
    <TabsContent value="indeterminate">
```luau
local Progress = Tab:ProgressBar({
    Title = "Syncing",
    Desc = "Please wait...",
    Indeterminate = true,
    IndeterminateText = "Syncing...",
    Speed = 1.2,
})
```
    </TabsContent>
</Tabs>

## Examples

### Custom value display

```luau
local Progress = Tab:ProgressBar({
	Title = "Storage",
	Desc = "Used space",
	Value = { Min = 0, Max = 100, Default = 72 },
	DisplayMode = "Value",
	Width = 220,
})
```

### Custom formatter

```luau
local Progress = Tab:ProgressBar({
	Title = "Bandwidth",
	Desc = "Live traffic",
	Value = { Min = 0, Max = 100, Default = 42 },
	ValueWidth = 84,
	Format = function(value, percentage, min, max)
		return string.format("%d%% of %d", math.floor(percentage + 0.5), max)
	end,
})
```
